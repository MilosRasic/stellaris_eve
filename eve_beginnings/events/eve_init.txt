namespace = eve

#Spawn Ni-Kunni as primitives
event = {
	id = eve.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

	immediate = {
		random_planet = {
			limit = {
				is_planet_class = pc_continental
				num_pops = 0
				not = {
					has_planet_flag = "named_planet"
				}
				is_moon = no
			}

			prevent_anomaly = yes

			create_species = {
				class = HUM
				portrait = gallente
				name = "Ni-Kunni"
				plural = "Ni-Kunni"
				adjective = "Ni-Kunni"
				name_list = "nikunni"
				homeworld = THIS
				traits = {
					trait = "trait_conformists"
				}
			}

			create_country = {
				name = "Ni-Kunni City States"
				authority = auth_oligarchic
				civics = {
					civic = civic_landed_nobility
					civic = civic_increasing_urbanization
				}
				species = last_created
				ethos = {
					ethic = "ethic_spiritualist"
				}
				flag = {
					icon = {
						category = "eve_online"
						file = "flag_eve_online_14.dds"
					}
					background= {
						category = "backgrounds"
						file = "00_solid.dds"
					}
					colors={
						"orange"
						"black"
						"null"
						"null"
					}
				}
				type = primitive
			}

			last_created_country = {
				set_graphical_culture = preindustrial_01
				set_country_flag = steam_age 
				set_primitive_age = steam_age
			}

			create_colony = {
				owner = last_created_country
				species = last_created
				ethos = owner
			}

			set_name = "Mishi"

			add_modifier = {
				modifier = "mineral_poor"
				days = -1
			}

			while = {
				count = 3
				random_tile = {
					limit = {
						has_blocker = no
						has_grown_pop = no
						has_growing_pop = no
					}
					create_pop = {
						species = last_created
					}
				}	
			}

			random_tile = {
				limit = { has_building = "building_colony_shelter" }
				remove_building = yes
			}

			while = {
				count = 3
				random_tile = {
					limit = { 
						has_blocker = no
						has_building = no
						has_grown_pop = yes
					}
					set_building = "building_primitive_farm"										
					set_deposit = d_farmland_deposit
				}
			}

			random_tile = {
				limit = { 
					has_blocker = no 
					has_building = no 
					has_grown_pop = yes
				}
				set_building = "building_primitive_factory"
				set_deposit = d_mineral_deposit					
			}

			while = {
				count = 3
				create_army = {
					name = "NAME_Industrial_Army"
					owner = last_created
					species = last_created
					type = "industrial_army"
				}
			}
		}
	}
}
